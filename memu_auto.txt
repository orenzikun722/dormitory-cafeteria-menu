# google colabгҒ®GCP_SERVICE_ACCOUNTгҒ«з„ЎзҗҶзҹўзҗҶservice_account.jsonгӮ’иІјгӮҠд»ҳгҒ‘гҒҰгҒ„гҒҫгҒҷ
# иҲһй¶ҙй«ҳе°Ӯе…¬ејҸгғӣгғјгғ гғҡгғјгӮёгҒ®еҜ® => йЈҹе Ӯй–ўдҝӮгҒ®гғӘгғігӮҜгӮ’гҒҷгҒ№гҒҰеҸ–еҫ—гҒ—гӮҝгғјгӮІгғғгғҲжңҲгҒ«еҪ“гҒҰгҒҜгҒҫгӮӢгғӘгғігӮҜгӮ’еҸ–еҫ—гҒ—гҒҫгҒҷ
# дёҖеәҰгғЎгғўгғӘгҒ«дҝқеӯҳ => pdfplumberгҒ«гҒқгҒ®гҒҫгҒҫжөҒгҒҷгҒ“гҒЁгҒ§и§ЈжһҗгҒ—гҒҰдҝқеӯҳгҒ—гҒҫгҒҷ
# гҒ“гҒ®д»•зө„гҒҝгӮ’дҪҝгҒҶгҒ“гҒЁгҒ§дёӢгҒ®иЁӯе®ҡгӮ’еӨүгҒҲгҒҰе®ҹиЎҢгҒҷгӮӢгҒ гҒ‘гҒ§дҝқеӯҳгҒ•гӮҢгҒҫгҒҷ


#гғЎгӮӨгғіиЁӯе®ҡ
month = 6
year = 2026



import requests
from bs4 import BeautifulSoup
import io
import pdfplumber
import gspread
from google.oauth2.service_account import Credentials
import datetime
import calendar
from google.colab import userdata

service_account_str = userdata.get('GCP_SERVICE_ACCOUNT')
json_path = '/content/service_account.json'
with open(json_path, 'w') as f:
    f.write(service_account_str)

dormitory_url = "https://www.maizuru-ct.ac.jp/campuslife/dormitory/"
notice_url = ""
response = requests.get(dormitory_url)
soup = BeautifulSoup(response.content, "html.parser")
for link in soup.select("#tab3")[0].find_all("li"):
  title = link.select('.entrytitle')[0].get_text()
  if "еӯҰеҜ®йЈҹе ӮзҢ®з«ӢиЎЁ" in title:
    if str(month) + "жңҲ" in title:
      notice_url = link.find('a').attrs['href']
if notice_url != "":
  response = requests.get(notice_url)
  soup = BeautifulSoup(response.content, "html.parser")
  pdf_url = soup.select('.overflow')[0].find('a').attrs['href']
  response = requests.get(pdf_url)
  if response.status_code == 200:
    with pdfplumber.open(io.BytesIO(response.content)) as pdf:
      week = ["жңҲ","зҒ«","ж°ҙ","жңЁ","йҮ‘","еңҹ","ж—Ҙ"]
      days = calendar.monthrange(year, month)[1]
      for day in range(1, days + 1):
        date = datetime.datetime.strptime(
          "2026е№ҙ"+str(month)+"жңҲ"+str(day)+"ж—Ҙ",
          "%Yе№ҙ%mжңҲ%dж—Ҙ"
        )
        target_date = "2026е№ҙ"+str(month)+"жңҲ"+str(day)+"ж—Ҙ("+week[date.weekday()]+")"
        for page in pdf.pages:
          table = page.extract_table()
          if not table:
            continue
          headers = table[0]
          try:
            date_col = headers.index(target_date)
          except ValueError:
            continue
          filtered_rows = []
          for row_idx, row in enumerate(table):
            if row_idx == 0:
              continue
            item = row[date_col]
            if item is None:
              continue
            item = item.strip()
            if item:
              if "\nA\n" in item:
                item = item.replace("\nA\n", "")
                item = "[ пјЎ ]\n" + item
              else:
                item = item.replace("A", "[ пјЎ ]\n")
              if "\nB\n" in item:
                item = item.replace("\nB\n", "")
                item = "[ пјў ]\n" + item
              else:
                item = item.replace("B", "[ пјў ]\n")
                filtered_rows.append(item)
            energy_indices = [
              i for i, item in enumerate(filtered_rows) if "пҪҙпҫҲпҫҷпҪ·пҫһпҪ°" in item
            ]
            if len(energy_indices) >= 3:
              breakfast_list = filtered_rows[0 : energy_indices[0]]
              lunch_list = filtered_rows[energy_indices[0] + 1 : energy_indices[1]]
              dinner_list = filtered_rows[energy_indices[1] + 1 : energy_indices[2]]
              breakfast = "\n".join(breakfast_list)
              lunch = "\n".join(lunch_list)
              dinner = "\n".join(dinner_list)
              if breakfast != "" and lunch != "" and dinner != "":
                scopes = ["https://www.googleapis.com/auth/cloud-platform", "https://www.googleapis.com/auth/drive", "https://www.googleapis.com/auth/spreadsheets"]
                creds = Credentials.from_service_account_file(
                  json_path,
                  scopes=scopes
                )
                client = gspread.authorize(creds)
                sheet = client.open("еҜ®йЈҹз®ЎзҗҶ").sheet1
                before_data = sheet.find(target_date)
                if before_data == None:
                  sheet.append_row([target_date, breakfast, lunch, dinner])
                else:
                  sheet.update_cell(before_data.row, before_data.col+1, breakfast)
                  sheet.update_cell(before_data.row, before_data.col+2, lunch)
                  sheet.update_cell(before_data.row, before_data.col+3, dinner)
                  print(target_date+"гҒ®иЁҳе…Ҙе®ҢдәҶ")
              else:
                if breakfast == "":
                  print(target_date+": жңқйЈҹгҒҢеҸ–еҫ—гҒ§гҒҚгҒҰгҒ„гҒҫгҒӣгӮ“")
                if lunch == "":
                  print(target_date+": жҳјйЈҹгҒҢеҸ–еҫ—гҒ§гҒҚгҒҰгҒ„гҒҫгҒӣгӮ“")
                if dinner == "":
                  print(target_date+": еӨ•йЈҹгҒҢеҸ–еҫ—гҒ§гҒҚгҒҰгҒ„гҒҫгҒӣгӮ“")

  else:
    print(f"PDFгҒ®еҸ–еҫ—гҒ«еӨұж•—гҒ—гҒҫгҒ—гҒҹгҖӮгӮ№гғҶгғјгӮҝгӮ№гӮігғјгғү: {response.status_code}")
else:
  print(f"жңҖж–°гҒ®гҒҠзҹҘгӮүгҒӣ5д»¶гҒӢгӮүзӣ®жЁҷгҒ®жңҲгҒ®pdfгӮ’еҸ–еҫ—гҒ§гҒҚгҒҫгҒӣгӮ“гҒ§гҒ—гҒҹ")
